```python
def test_output_format_detection():
    """Web route should detect input format from filename and output matching format."""
    import os

    # Simulate the format detection logic that will be added to app.py
    test_cases = [
        ('photo.jpg', 'JPEG', 'image/jpeg'),
        ('photo.jpeg', 'JPEG', 'image/jpeg'),
        ('photo.JPG', 'JPEG', 'image/jpeg'),
        ('image.png', 'PNG', 'image/png'),
        ('image.PNG', 'PNG', 'image/png'),
        ('image.webp', 'WEBP', 'image/webp'),
        ('noext', 'JPEG', 'image/jpeg'),         # fallback
        ('weird.bmp', 'JPEG', 'image/jpeg'),      # unknown ext → fallback
    ]

    fmt_map = {'.png': ('PNG', 'image/png'), '.webp': ('WEBP', 'image/webp')}

    for filename, expected_fmt, expected_mime in test_cases:
        ext = os.path.splitext(filename)[1].lower()
        save_fmt, mime = fmt_map.get(ext, ('JPEG', 'image/jpeg'))
        assert save_fmt == expected_fmt, f"{filename}: expected fmt {expected_fmt}, got {save_fmt}"
        assert mime == expected_mime, f"{filename}: expected mime {expected_mime}, got {mime}"

    print("[+] test_output_format_detection: PASS")
    return True
```
